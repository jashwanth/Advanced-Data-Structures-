#include <algorithm>
#include <vector>
#include <iostream>

bool forward(int lhs, int rhs){
  if(lhs < rhs){
    return true;
  }
  return false;
}

bool reverse(int lhs, int rhs){
  if(rhs < lhs){
    return true;
  }
  return false;
}

bool even(int lhs, int rhs){
  if(lhs % 2 == 0 && rhs % 2 == 0){
    return true;
  }
  return false;
}

void print(std::vector<int>& numbers, const char * label){
  std::cout << label << ": [";
  for(size_t i = 0; i < numbers.size(); ++i){
    std::cout << numbers[i];
    if(i < numbers.size() - 1){
      std::cout << ",";
    }
  }
  std::cout << "]" << std::endl;
}

int main(int argc, char * argv[]){
  std::vector<int> numbers;
  numbers.push_back(10);
  numbers.push_back(20);
  numbers.push_back(0);
  numbers.push_back(15);
  numbers.push_back(9);

  std::sort(numbers.begin(), numbers.end(), forward);
  print(numbers, "forward");
  std::sort(numbers.begin(), numbers.end(), reverse);
  print(numbers, "reverse");
  std::sort(numbers.begin(), numbers.end(), even);
  print(numbers, "evenodd");

  return 0;
}
