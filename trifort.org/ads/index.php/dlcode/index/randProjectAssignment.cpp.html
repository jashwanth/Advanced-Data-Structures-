#include <cstdlib>
#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <ctime>
#include <algorithm>

std::vector<std::vector<std::string> > assign(int num_people, std::vector<std::string> names)
{
  std::vector<int> people;
  for(int i = 0; i < names.size(); ++i){
    people.push_back(i % num_people);
  }

  std::random_shuffle(people.begin(), people.end()); 

  std::vector<std::vector<std::string> > ret;
  ret.resize(num_people);

  for(int i = 0; i < people.size(); ++i){
    int id = people[i];
    ret[id].push_back(names[i]);
  }
  
  return ret;
}

std::string toString(int i)
{
  std::ostringstream oss;
  oss << i;
  return oss.str();
}

int main(int argc, char * argv[])
{

  std::vector<std::string> names;
  for(int i = 0; i < 10; ++i){
    names.push_back(toString(i));
  }

  std::vector<std::vector<std::string> > groups = assign(3, names);
  for(size_t i = 0; i < groups.size(); ++i){
    std::cout << "group: " << i << std::endl;
    std::vector<std::string> curr = groups[i];
    for(size_t j = 0; j < curr.size(); ++j){
      std::cout << "  name: " << curr[j] << std::endl;
    }
  }

  //group: 0

  //group: 1
  //  name: 0
  //  name: 2
  //  name: 7
  //group: 2
  //  name: 3

  //  name: 6

  return 0;
}
