#include <iostream>
#include <map>
#include <string>

int main(int argc, char *argv[]){

  std::multimap<std::string, int> course_catalog;
  course_catalog.insert(std::pair<std::string, int>("cis", 252));
  course_catalog.insert(std::pair<std::string, int>("cis", 275));
  course_catalog.insert(std::pair<std::string, int>("cis", 300));
  course_catalog.insert(std::pair<std::string, int>("cse", 283));
  course_catalog.insert(std::pair<std::string, int>("cse", 382));
  course_catalog.insert(std::pair<std::string, int>("cse", 484));
  course_catalog.insert(std::pair<std::string, int>("ele", 231));
  course_catalog.insert(std::pair<std::string, int>("ele", 232));
  course_catalog.insert(std::pair<std::string, int>("ele", 312));

  std::multimap<std::string, int>::iterator iter;
  iter = course_catalog.find("cse");
  std::cout << "courses in cse: " << std::endl;
  while(true){
    std::pair<std::string, int> item = *iter;
    if(item.first == "cse"){
      std::cout << item.first << item.second << std::endl;
      ++iter;
    } else {
      break;
    }
  }
  
  //prints:
  //  courses in cse:
  //  cse283
  //  cse382
  //  cse484

  return 0;
}
