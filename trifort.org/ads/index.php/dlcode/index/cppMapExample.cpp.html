#include <iostream>
#include <map>
#include <string>

int main(int argc, char *argv[]){

  std::map<std::string, std::string> dictionary;
  dictionary["array"] = "a systematic arrangement of objects, usually in rows and columns";
  dictionary["vector"] = "an automatically resizable array";
  dictionary["queue"] = "a container where the next element is given in FIFO order";
  dictionary["stack"] = "a container where the next element is given in LIFO order";

  std::cout << "an array is: " << dictionary["array"] << std::endl;
  std::cout << std::endl;  

  std::cout << "the dictionary contents: " << std::endl;
  std::map<std::string, std::string>::iterator iter;
  for(iter = dictionary.begin(); iter != dictionary.end(); ++iter){
    std::pair<std::string, std::string> item = *iter;
    std::cout << item.first << ": " << item.second << std::endl;
  }
  std::cout << std::endl;

  dictionary.erase("array");
  std::cout << "after erasing 'array', ";
  if(dictionary.find("array") != dictionary.end()){
    std::cout << "dictionary contains array" << std::endl;
  } else {
    std::cout << "dictionary does not contain array" << std::endl;
  }
  
  //prints:
  //  an array is: a systematic arrangement of objects, usually in rows and columns
  //
  //  the dictionary contents: 
  //  array: a systematic arrangement of objects, usually in rows and columns
  //  queue: a container where the next element is given in FIFO order
  //  stack: a container where the next element is given in LIFO order
  //  vector: an automatically resizable array
  //
  //  after erasing 'array', dictionary does not contain array

  return 0;
}
