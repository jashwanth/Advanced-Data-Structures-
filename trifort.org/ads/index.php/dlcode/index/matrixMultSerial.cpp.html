#include "UnixStopwatch.h"
#include <iostream>

int ** newMatrix(int size){

  int ** ret = new int*[size];
  for(int i = 0; i < size; ++i){
    ret[i] = new int[size];
    for(int j = 0; j < size; ++j){
      ret[i][j] = i * j;
    }
  }
  
  return ret;
}

void multiply(int ** a, int ** b, int ** c, int size){
  for(int i = 0; i < size; ++i){
    for(int j = 0; j < size; ++j){
      int sum = 0;
      for(int k = 0; k < size; ++k){
        sum += a[i][k] * b[k][j];
      }
      c[i][j] = sum;
    }
  }
}

void multiplyTranspose(int ** a, int ** b, int ** c, int size){
  for(int i = 0; i < size; ++i){
    for(int j = 0; j < size; ++j){
      int sum = 0;
      for(int k = 0; k < size; ++k){
        sum += a[i][k] * b[j][k];
      }
      c[i][j] = sum;
    }
  }
}

int main(int argc, char * argv[]){

  int size = 1024;
  int ** a = newMatrix(size);
  int ** b = newMatrix(size);
  int ** c = newMatrix(size);

  UnixStopwatch watch;
  multiply(a, b, c, size);
  watch.stop();

  std::cout << "regular multiply elapsed ms: " << watch.getTime() << std::endl;

  watch.start();
  multiplyTranspose(a, b, c, size);
  watch.stop();

  std::cout << "transpose multiply elapsed ms: " << watch.getTime() << std::endl;

  //prints:
  //  regular multiply elapsed ms: 29622
  //  transpose multiply elapsed ms: 16640

  return 0;
}
