#include <queue>
#include <iostream>
#include <functional>

int main(int argc, char *argv[]){
 
  std::priority_queue<int> max_queue;
  std::priority_queue<int, std::vector<int>, std::greater<int> > min_queue;
  //std::priority_queue<int> uses std::vector<int> and std::less<int> by default

  min_queue.push(3);
  min_queue.push(2);
  min_queue.push(1);

  max_queue.push(3);
  max_queue.push(2);
  max_queue.push(1);

  std::cout << "top of max_queue: " << max_queue.top() << std::endl;
  //prints: top of max_queue: 3
  max_queue.pop();

  std::cout << "top of min_queue: " << min_queue.top() << std::endl;
  //prints: top of min_queue: 1
  min_queue.pop();
  return 0;
}
