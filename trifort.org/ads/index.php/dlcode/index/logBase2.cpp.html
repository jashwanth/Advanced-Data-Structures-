#include <iostream>

int logBase2(int n){
  int ret = 0;
  while(n > 1){
    n /= 2;
    ret++;
  }
  return ret;
}

int main(int argc, char * argv[]){

  for(int i = 0; i <= 32; ++i){
    int log2 = logBase2(i);
    std::cout << "the log base 2 of (" << i << ") = " << log2 << std::endl;
  }

  return 0;
}
