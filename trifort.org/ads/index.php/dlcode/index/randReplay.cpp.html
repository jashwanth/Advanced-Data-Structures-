#include <cstdlib>
#include <iostream>

void printRandomNumbers()
{
  std::cout << "five random numbers between 0 and 10:" << std::endl;
  for(int i = 0; i < 5; ++i){
    std::cout << rand() % 10 << " ";
  }
  std::cout << std::endl;
}

int main(int argc, char * argv[]){

  int seed = 20;

  srand(seed);
  printRandomNumbers();

  srand(seed);
  printRandomNumbers();

  //prints:
  //
  //  five random numbers between 0 and 10:
  //  1 8 7 9 6 
  //
  //  five random numbers between 0 and 10:
  //  1 8 7 9 6 

  return 0;
}
