BTreeNode * btreeSearch(int key, BTreeNode * curr){
  if(curr != NULL){
    int i;
    //seek the the closest key
    for(i = 0; i < curr->sizeKeys() && curr->getKey(i) < key; ++i);

    if(i >= curr->sizeKeys() || curr->getKey(i) > key){
      //if the above for loop broke because the key was not found, recurse
      return btreeSearch(key, curr->getPointer(i));
    } else {
      //else we found the node
      return curr;
    }
  } else {
    return NULL;
  }
}
