void rotateRight(TreeNode * gr, TreeNode * par, TreeNode * ch){
  //grandparent child becomes ch  
  if(gr->getRight() == par){
    gr->setRight(ch);
  } else {
    gr->setLeft(ch);
  }

  TreeNode * ch_right = ch->getRight();
  par->setLeft(ch_right);
  ch->setRight(par);
}

void rotateLeft(TreeNode * gr, TreeNode * par, TreeNode * ch){
  //grandparent child becomes ch  
  if(gr->getRight() == par){
    gr->setRight(ch);
  } else {
    gr->setLeft(ch);
  }
  
  TreeNode * ch_left = ch->getLeft();
  par->setRight(ch_left);
  ch->setLeft(par);
}
