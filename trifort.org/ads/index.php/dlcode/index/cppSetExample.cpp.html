#include <iostream>
#include <set>

int main(int argc, char * argv[]){

  std::set<int> set1;
  set1.insert(20);
  set1.insert(10);
  set1.insert(50);
  set1.insert(23);
  set1.insert(20);

  std::set<int>::iterator iter;
  std::cout << "items in set1: " << std::endl;
  for(iter = set1.begin(); iter != set1.end(); ++iter){
    std::cout << "  " << *iter << std::endl;
  }
  //prints:
  //  items in set1:
  //    10
  //    20
  //    23
  //    50

  return 0;
}
