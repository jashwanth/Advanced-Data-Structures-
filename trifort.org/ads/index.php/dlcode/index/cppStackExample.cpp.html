#include <stack>
#include <iostream>

int main(int argc, char * argv[]){

  std::stack<int> int_stack;
  int_stack.push(1);
  int_stack.push(2);

  //in the std library top and pop are two seperate methods
  std::cout << int_stack.top() << std::endl;
  int_stack.pop();

  int_stack.push(3);
  int_stack.push(4);
  int_stack.push(5);

  return 0;
}
