#include <iostream>
#include <fstream>

void computeChecksum(std::string filename)
{
  char checksum = 0;
  char prev_checksum = 0;
  int rollovers = 0;
  std::ifstream fin(filename.c_str(), std::ios::binary);
  while(fin.good()){
    char c;
    fin.get(c);
    checksum += c;
    if(checksum < prev_checksum){
      rollovers++;
    }
    prev_checksum = checksum;
  }
  std::cout << "checksum for " << filename << ": " << (int) checksum;
  std::cout << " (" << rollovers << " rollovers)" << std::endl;
}

int main(int argc, char * argv[]){
 
  computeChecksum("output.bin");
  computeChecksum("output.txt");

  //prints:
  //  checksum for output.bin: 10 (0 rollovers)
  //  checksum for output.txt: 112 (4 rollovers)
  return 0;
}
