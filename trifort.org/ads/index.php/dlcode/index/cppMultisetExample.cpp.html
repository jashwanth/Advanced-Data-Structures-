#include <iostream>
#include <set>

int main(int argc, char * argv[]){

  std::multiset<int> mset1;
  mset1.insert(20);
  mset1.insert(10);
  mset1.insert(50);
  mset1.insert(23);
  mset1.insert(20);

  std::multiset<int>::iterator iter;
  std::cout << "items in mset1: " << std::endl;
  for(iter = mset1.begin(); iter != mset1.end(); ++iter){
    std::cout << "  " << *iter << std::endl;
  }
  //prints:
  //  items in mset1:
  //    10
  //    20
  //    20
  //    23
  //    50

  return 0;
}
