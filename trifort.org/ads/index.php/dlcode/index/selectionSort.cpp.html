#include <iostream>
#include <vector>

void swap(std::vector<int>& a, int index1, int index2){
  int lhs = a[index1];
  int rhs = a[index2];
  a[index1] = rhs;
  a[index2] = lhs;
}

void selectionSort(std::vector<int>& a){
  size_t j, least;
  for(size_t i = 0; i < (a.size() - 1); ++i){
    for(j = i + 1, least = i; j < a.size(); ++j){
      if(a[j] < a[least]){
        least = j;
      }
    }
    swap(a, least, i);
  }
}

int main(int argc, char * argv[]){

  std::vector<int> array;
  array.push_back(10);
  array.push_back(5);
  array.push_back(15);
  array.push_back(7);
  array.push_back(1);

  selectionSort(array);
  
  for(size_t i = 0; i < array.size(); ++i){
    std::cout << array[i] << " ";
  }
  std::cout << std::endl;
  //prints: 1 5 7 10 15 

  return 0;
}
