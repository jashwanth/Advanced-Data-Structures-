#include <iostream>

class TreeNode {
public:
  TreeNode(int value);
  int getValue();
  void setLeft(TreeNode * left);
  void setRight(TreeNode * right);
  TreeNode * getLeft();
  TreeNode * getRight();
private:
  int m_Value;
  TreeNode * m_Left;
  TreeNode * m_Right;
};

TreeNode::TreeNode(int value)
  : m_Value(value), m_Left(NULL), m_Right(NULL)
{
}

int TreeNode::getValue(){
  return m_Value;
}

void TreeNode::setLeft(TreeNode * left){
  m_Left = left;
}

void TreeNode::setRight(TreeNode * right){
  m_Right = right;
}

TreeNode * TreeNode::getLeft(){
  return m_Left;
}

TreeNode * TreeNode::getRight(){
  return m_Right;
}

TreeNode * makeTree(){
  TreeNode * root = new TreeNode(20);
  TreeNode * left0 = new TreeNode(10);
  TreeNode * right0 = new TreeNode(40);
  root->setLeft(left0);
  root->setRight(right0);
  TreeNode * leftleft1 = new TreeNode(5);
  TreeNode * leftright1 = new TreeNode(15);
  left0->setLeft(leftleft1);
  left0->setRight(leftright1);
  leftleft1->setLeft(new TreeNode(2));
  leftright1->setLeft(new TreeNode(11));
  leftright1->setRight(new TreeNode(17));
  TreeNode * rightright1 = new TreeNode(45);
  right0->setRight(rightright1);
  rightright1->setLeft(new TreeNode(43));
  rightright1->setRight(new TreeNode(53));
  return root;
}

//Robert Sedgewick, Algorithms in C++, Parts 1-4, page 250.
int countNodes(TreeNode * curr){
  if(curr == NULL){
    return 0;
  }
  return countNodes(curr->getLeft()) + countNodes(curr->getRight()) + 1;
}

//Robert Sedgewick, Algorithms in C++, Parts 1-4, page 250.
int calcHeight(TreeNode * curr){
  if(curr == NULL){
    return 0;
  }
  int left_height = calcHeight(curr->getLeft());
  int right_height = calcHeight(curr->getRight());
  if(left_height > right_height){
    return left_height + 1;
  } else {
    return right_height + 1;
  }
}

int main(int argc, char * argv[]){

  TreeNode * root = makeTree();
  std::cout << "num nodes: " << countNodes(root) << std::endl; 
  std::cout << "height: " << calcHeight(root) << std::endl;
  //prints:
  //  num nodes: 11
  //  height: 4

  return 0;
}
